#!/bin/bash

# --- KONFIGURACJA SIECI ---
PIHOLE_DIR="$HOME/pihole"
PASSWORD="twoje_haslo_tutaj"
TIMEZONE="Europe/Warsaw"

# Adres IP Twojego FunBoxa
ROUTER_IP="192.168.1.1"
# Zakres adresów IP dla pozostałych urządzeń
DHCP_START="192.168.1.10"
DHCP_END="192.168.1.250"

# --- SEKCOJA: STAŁE ADRESY IP (REZERWACJE) ---
# Format: "MAC_ADRES,ADRES_IP,NAZWA"
# MAC adres znajdziesz na naklejce urządzenia lub w panelu FunBoxa
STATYCZNE_IP=(
    "AA:BB:CC:DD:EE:FF,192.168.1.5,MojKomputer"
    "11:22:33:44:55:66,192.168.1.6,Drukarka"
    "00:11:22:33:44:55,192.168.1.7,Telewizor"
)

echo "--- 1. Przygotowanie systemu ---"
sudo usermod -aG docker $USER
sudo sed -i 's/#DNSStubListener=yes/DNSStubListener=no/' /etc/systemd/resolved.conf
sudo sed -i 's/DNSStubListener=yes/DNSStubListener=no/' /etc/systemd/resolved.conf
sudo rm /etc/resolv.conf
echo "nameserver 8.8.8.8" | sudo tee /etc/resolv.conf
sudo systemctl restart systemd-resolved

echo "--- 2. Tworzenie folderów ---"
mkdir -p "$PIHOLE_DIR/etc-pihole"
mkdir -p "$PIHOLE_DIR/etc-dnsmasq.d"

echo "--- 3. Generowanie statycznych rezerwacji IP ---"
# Czyścimy stary plik i wpisujemy nowe rezerwacje
STATIC_CONF="$PIHOLE_DIR/etc-dnsmasq.d/04-static-dhcp.conf"
echo "# Plik wygenerowany automatycznie - statyczne IP" > "$STATIC_CONF"

for entry in "${STATYCZNE_IP[@]}"; do
    echo "dhcp-host=$entry" >> "$STATIC_CONF"
done

echo "--- 4. Generowanie docker-compose.yml ---"
cat <<EOF > "$PIHOLE_DIR/docker-compose.yml"
services:
  pihole:
    container_name: pihole
    image: pihole/pihole:latest
    network_mode: host
    environment:
      TZ: '$TIMEZONE'
      FTLCONF_webserver_api_password: '$PASSWORD'
      FTLCONF_dns_listeningMode: 'ALL'
      FTLCONF_dhcp_active: 'true'
      FTLCONF_dhcp_range: '$DHCP_START,$DHCP_END,24h'
      FTLCONF_dhcp_router: '$ROUTER_IP'
    dns:
      - 127.0.0.1
      - 8.8.8.8
    volumes:
      - './etc-pihole:/etc/pihole'
      - './etc-dnsmasq.d:/etc/dnsmasq.d'
    cap_add:
      - NET_ADMIN
      - SYS_TIME
      - SYS_NICE
    restart: unless-stopped
EOF

echo "--- 5. Uruchamianie Pi-hole ---"
cd "$PIHOLE_DIR"
sudo docker compose down
sudo docker compose up -d

echo "-------------------------------------------------------"
echo "GOTOWE! Rezerwacje IP zostały dodane."
echo "Skonfigurowane urządzenia:"
for entry in "${STATYCZNE_IP[@]}"; do echo " - $entry"; done
echo "-------------------------------------------------------"
echo "Pamiętaj o wyłączeniu DHCP w FunBoxie!"
